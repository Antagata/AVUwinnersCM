import subprocess, sys, shutil
from datetime import datetime

print("\n" + "="*60)
print("PUSHING UPDATES TO GITHUB PAGES")
print("="*60)

try:
    # Apply fixes
    print("\n[1/3] Applying bug fixes...")
    subprocess.run([sys.executable, "fix_dashboard_bugs.py"], check=True)

    # Copy to index
    print("[2/3] Copying to index.html...")
    shutil.copy2("dashboard.html", "index.html")

    # Git operations
    print("[3/3] Pushing to GitHub...")
    subprocess.run(["git", "add", "dashboard.html", "index.html"], check=True)

Q: Git push fails?
A: Run: git pull origin master
   Then try pushing again
